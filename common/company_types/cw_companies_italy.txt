		has_interest_marker_in_region = region_italy
	}
	
	attainable = {

				}
			}
		}
	}

		any_scope_state = {

			is_incorporated = yes
			any_scope_building = {
				OR = {
					is_building_type = building_car_factory #汽车制造厂
					is_building_type = building_motor_industry #动力机械厂
					is_building_type =	 building_electrics_industry #电子产品厂
				}
				level >= 3
			}
		}
	}
	
	prosperity_modifier = {
		military_formation_movement_speed_mult = 0.2
		unit_army_defense_mult = 0.1
	}
