ajinomoto_company = {
	icon = "gfx/interface/icons/company_icons/historical_company_icons/ajinomoto.dds"
	background = "gfx/interface/icons/company_icons/company_backgrounds/comp_illu_manufacturing_light.dds"
	
	flavored_company = yes
	
	building_types = {  
		building_food_industry #食品厂
		building_sugar_factory #制糖厂
		building_sugar_plantation #糖料作物种植园
		building_pharmaceutical_factory #医疗用品厂
	}
	
	potential = {
			has_interest_marker_in_region = region_japan
	}
	
	attainable = {
		hidden_trigger = { # Hide attainable if it's always fulfilled by possible
			any_scope_state = {
					region = sr:region_japan
			}
		}
	}			
	
	possible = {
		any_scope_state = {
				region = sr:region_japan
			is_incorporated = yes
			any_scope_building = {
				OR = {
					is_building_type = building_food_industry #食品厂
					is_building_type = building_sugar_factory #制糖厂
					is_building_type = building_sugar_plantation #糖料作物种植园
					is_building_type = building_pharmaceutical_factory #医疗用品厂
				}
				level >= 5
			}
		}
	}
	
	prosperity_modifier = {
		state_radicals_from_sol_change_mult = -0.1 #生活水平降低所带来的的激进派
		unit_morale_recovery_mult = 0.15
	}	
	
	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
