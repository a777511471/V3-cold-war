
TPG_clothes_pop_trigger = {  #太平天国通用
exists = scope:pop
scope:pop = {
exists = c:TPG owner = c:TPG
}
}

TPG_chairman_clothes_trigger = { #太平天国主席
exists = scope:character
scope:character = {
is_ruler = yes
owner = { has_law = law_type:law_council_republic AND = { exists = c:TPG owner = c:TPG } }
}
}
