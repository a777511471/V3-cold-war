
SRC_clothes_pop_trigger = {  #中华苏维埃通用
exists = scope:pop
scope:pop = {
exists = c:SRC owner = c:SRC
}
}

SRC_chairman_clothes_trigger = { #中华苏维埃主席
exists = scope:character
scope:character = {
is_ruler = yes
owner = { has_law = law_type:law_council_republic AND = { exists = c:SRC owner = c:SRC } }
}
}
