liaoshenzhanyi_button_key = {
	name = "liaoshenzhanyi_button_key"
	desc = "liaoshenzhanyi_button_key_DESC"

	visible = { always = yes }
	
	#Country scope
	ai_chance = {
		base = 100
	}

	possible = {
		custom_tooltip = {
			text = liaoshenzhanyi_button_tt
			hidden_trigger = {
				var:liaoshenzhanyi_var = 0
				NOT = {
					any_scope_war = {
						any_scope_front = {
							any_scope_state = {
								OR = {
									state_region = s:STATE_SHENGJING
									state_region = s:STATE_SOUTHERN_MANCHURIA
									state_region = s:STATE_NORTHERN_MANCHURIA	
								}
							}
						}
					}
				}
			}
		}
	}

	effect = {
		change_variable = { name = liaoshenzhanyi_var add = 1 }
		trigger_event = { id = sandazhanyi.1 days = 30 }
	}
}
