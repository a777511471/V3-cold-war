	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

    complete = {
        owns_entire_state_region = STATE_CEYLON
        OR = {
            owns_entire_state_region = STATE_LHASA
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_LHASA
            }
        }
        OR = {
            owns_entire_state_region = STATE_NGARI
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_NGARI
            }
        }
        OR = {
            owns_entire_state_region = STATE_WESTERN_AFGHANISTAN
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_WESTERN_AFGHANISTAN
            }
        }
        OR = {
            owns_entire_state_region = STATE_EASTERN_AFGHANISTAN
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_EASTERN_AFGHANISTAN
            }
        }
        OR = {
            owns_entire_state_region = STATE_KACHIN
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_KACHIN
            }
        }
        OR = {
            owns_entire_state_region = STATE_PEGU
            any_country = {
                is_subject_of = c:BHT
                owns_entire_state_region = STATE_PEGU
            }
        }      
    }
    on_complete = {
		add_modifier = {
			name = bht_expansion_second_modifier
            months = -1
		}
		trigger_event = {
			id = bht_expansion_second.3
		}
	}
    fail = {

				text = lost_war_tt

					AND = {
						has_variable = recently_had_war

							has_variable = recently_won_war
						}
					}
				}
			}
    }
    on_fail = {
		trigger_event = {
			id = bht_expansion_second.2
		}
	}
    weight = 1000
    should_be_pinned_by_default = yes
