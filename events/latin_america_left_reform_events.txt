latin_america_left_reform.1 = {
	type = country_event
	placement = root
	
	title = latin_america_left_reform.1.t
	desc = latin_america_left_reform.1.d
	flavor = latin_america_left_reform.1.f
	
	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	
	duration = 3 

	trigger = {
		any_scope_character = {
			is_ruler = yes
 			OR = {
				ideology = ideology_social_democrat
				ideology = ideology_communist
				ideology = ideology_anarchist
			}           
		}
		NOT = {
			has_variable = latin_america_left_reform_triggered
		}
        OR = {
            c:CHL ?= this
            c:PEU ?= this
            c:MEX ?= this
            c:ARG ?= this
            c:CLM ?= this
            c:VNZ ?= this
            c:ECU ?= this
            c:PRG ?= this
            c:URU ?= this
            c:NIC ?= this
            c:HON ?= this
            c:PNM ?= this
            c:COS ?= this
            c:BOL ?= this
            c:HAI ?= this
        }
	}

    immediate = {
        set_variable = latin_america_left_reform_triggered
    }  
    option = {
		name = latin_america_left_reform.1.a
		add_journal_entry = {
			type = je_latin_america_left_reform
		}
		default_option = yes
    }
    option = {
		name = latin_america_left_reform.1.b
		default_option = no
    }
}
